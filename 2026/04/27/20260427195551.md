+				return isInBounds(value);
+			}

+			if (usePrefix) {
+				// only allow auto switching if using number prefix
+				return autoSwitchFormat(text);
+			}
+			return false;
+		}

+				}

+				if (value != null && isInBounds(value)) {
+					currentFormat = format;
+					textField.setFormat(format);
+					return true;
+				}
+			}
+			return false;
+		}
